**前序即根在前面——根左右
中序即根在中间——左根右
后序即根在后面——左右根
永远是左在右前面**

```go
func inorderTraversal(root *TreeNode) []int {
	res := []int{}
	inorder(root, &res)
	return res
}

func inorder(root *TreeNode, res *[]int)  {
	if root == nil {
		return
	}
	inorder(root.Left, res)
	*res = append(*res, root.Val)
	inorder(root.Right, res)
}
```
