**前序即根在前面——根左右
中序即根在中间——左根右
后序即根在后面——左右根
永远是左在右前面**

```go
func preorderTraversal(root *TreeNode) []int {
	res := []int {}
	preorder(root, &res)
	return res
}

func preorder(root *TreeNode, res *[]int)  {
	if root == nil {
		return
	}
	*res = append(*res, root.Val)
	preorder(root.Left, res)
	preorder(root.Right, res)
}
```
